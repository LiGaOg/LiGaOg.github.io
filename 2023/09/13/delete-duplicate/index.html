删除有序数组重复项I

https://leetcode.cn/problems/remove-duplicates-from-sorted-array/

经典的双指针模板

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {

        int p = 0;
        for (int i = 0, j = 0; i < nums.size(); i ++) {
            while (j < nums.size() && nums[i] == nums[j]) j++;
            nums[p ++] = nums[i];
            i = j - 1;
        }
        return p;
    }
};
